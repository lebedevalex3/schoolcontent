\worksheettitle
  {Домашняя работа: ответы}
  {\modulemark{M04-H} Версия учителя}

\begin{teacherbox}[Назначение]
  Работа проверяет чтение, согласование названий классов, запись неполных групп
  и сохранение нулевого промежуточного класса.
\end{teacherbox}

\section{Ответы}

\begin{teacherbox}[1. Прочитайте числа]
  \begin{enumerate}
    \item семь тысяч двадцать один;
    \item тридцать две тысячи четыре;
    \item шестьсот четырнадцать тысяч восемьдесят;
    \item двадцать один миллион две тысячи пятнадцать;
    \item сто восемь миллионов двести четыре.
  \end{enumerate}
\end{teacherbox}

\begin{teacherbox}[2. Выберите формы]
  Сорок одна тысяча; пятьдесят две тысячи; сто тринадцать тысяч; двадцать
  четыре миллиона.
\end{teacherbox}

\begin{teacherbox}[3. Запишите цифрами]
  \begin{enumerate}
    \item \(43\,008\);
    \item \(702\,050\);
    \item \(12\,005\,009\);
    \item \(80\,000\,032\);
    \item \(300\,002\,104\).
  \end{enumerate}
\end{teacherbox}

\newpage
\begin{teacherbox}[4. Исправьте ошибки]
  \begin{enumerate}
    \item \(6\,000\,014\): потерян нулевой класс тысяч.
    \item «Сорок пять миллионов семьдесят два»: класс тысяч нулевой, группа
      \(072\) относится к классу единиц.
    \item \(31\,005\): группа класса единиц должна содержать три цифры.
  \end{enumerate}
\end{teacherbox}

\begin{teacherbox}[5. Итог]
  \begin{enumerate}
    \item сорок два миллиона одна тысяча шесть;
    \item \(60\,000\,073\);
    \item группа \(000\) сохраняет нулевой класс тысяч, а \(073\) сохраняет три
      позиции класса единиц.
  \end{enumerate}
\end{teacherbox}

\begin{resultbox}[Критерии проверки]
  \begin{itemize}
    \item \textbf{Освоено:} не менее 15 из 17 элементов выполнены верно,
      итоговое объяснение связано с классами.
    \item \textbf{Точечная доработка:} одна повторяющаяся грамматическая или
      позиционная ошибка; дать два аналогичных примера.
    \item \textbf{M04-R:} потерян нулевой класс, группы не дополнены нулями или
      ученик не различает \(005\) и количество \(5\).
  \end{itemize}
\end{resultbox}
