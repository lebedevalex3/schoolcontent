\studentnameblock

\begin{notebox}[Опора]
  Сначала заполните отдельные группы. В каждом непервом классе должно быть
  ровно три цифры. Неназванный класс записывайте как (000).
\end{notebox}

\begin{center}
\renewcommand{\arraystretch}{1.7}
\begin{tabular}{c|c|c}
  \toprule
  миллионы & тысячи & единицы \\
  \midrule
  & & \\
  \bottomrule
\end{tabular}
\end{center}

\begin{practicebox}[По шагам]
  Запишите: \emph{тридцать миллионов четыре тысячи семь}.
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item Миллионы: \answerblank[20mm].
    \item Тысячи: число 4, группа \answerblank[20mm].
    \item Единицы: число 7, группа \answerblank[20mm].
    \item Всё число: \answerblank[55mm].
  \end{enumerate}
\end{practicebox}

\begin{practicebox}[Теперь самостоятельно]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item Запишите: \emph{пять миллионов двенадцать}: \answerblank[50mm].
    \item Прочитайте $62\,000\,405$: \answerblank[82mm].
    \item Запишите: \emph{сто один миллион девять тысяч сорок}:
      \answerblank[55mm].
  \end{enumerate}
\end{practicebox}

\begin{checkpointbox}[Повторная проверка]
  Запишите цифрами: \emph{сорок два миллиона пять тысяч девять}:
  \answerblank[55mm]. Обведите каждую группу из трёх цифр.
\end{checkpointbox}
