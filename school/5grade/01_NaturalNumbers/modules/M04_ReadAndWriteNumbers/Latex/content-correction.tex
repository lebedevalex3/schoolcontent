\begin{notebox}[Опора]
  Сначала заполните группы классов. Все группы, кроме крайней левой, должны
  содержать по три цифры. Неназванный промежуточный класс записывайте как
  \(000\).
\end{notebox}

\section{Шаг 1. Читаем по классам}

\begin{problembox}[Образец]
  \[
    21\,|\,002\,|\,014.
  \]
  Читаем: «двадцать один миллион две тысячи четырнадцать».
  Нулевые позиции внутри групп не произносим.
\end{problembox}

\begin{practicebox}[Прочитайте]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(62\,000\,405\): \answerblank[86mm].
    \item \(31\,004\,008\): \answerblank[80mm].
    \item \(12\,012\,040\): \answerblank[80mm].
  \end{enumerate}
\end{practicebox}

\begin{practicebox}[Выберите правильную форму]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(21\,000\): двадцать \answerblank[14mm] тысяча.
    \item \(32\,000\): тридцать \answerblank[14mm] тысячи.
    \item \(114\,000\): сто четырнадцать \answerblank[18mm].
  \end{enumerate}
\end{practicebox}

\section{Шаг 2. Записываем по классам}

\begin{problembox}[По шагам]
  Запишите: \emph{тридцать миллионов четыре тысячи семь}.
  \begin{center}
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{c|c|c}
    \toprule
    миллионы & тысячи & единицы \\
    \midrule
    \answerblank[16mm] & \answerblank[16mm] & \answerblank[16mm] \\
    \bottomrule
  \end{tabular}
  \end{center}
  Всё число: \answerblank[48mm].
\end{problembox}

\begin{practicebox}[Запишите самостоятельно]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item пять миллионов двенадцать: \answerblank[48mm].
    \item сто один миллион девять тысяч сорок: \answerblank[48mm].
    \item семнадцать миллионов шесть: \answerblank[48mm].
  \end{enumerate}
\end{practicebox}

\begin{warningbox}[Исправьте ошибки]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(8\,5\,42\) вместо \emph{восемь миллионов пять тысяч сорок два}.
    \item \(17\,6\) вместо \emph{семнадцать миллионов шесть}.
  \end{enumerate}
  Верные записи и причины:
  \writinglines{3}
\end{warningbox}
