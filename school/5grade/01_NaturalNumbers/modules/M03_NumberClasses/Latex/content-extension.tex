\begin{problembox}[Четыре класса]
  Рассмотрите число \(7\,|\,304\,|\,052\,|\,018\).
  \begin{center}
  \renewcommand{\arraystretch}{1.4}
  \begin{tabular}{c|cccc}
    \toprule
    класс & миллиарды & миллионы & тысячи & единицы \\
    \midrule
    группа & 7 & 304 & 052 & 018 \\
    количество единиц класса & \answerblank[12mm] & \answerblank[14mm] &
      \answerblank[14mm] & \answerblank[14mm] \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{problembox}

\begin{practicebox}[1. Разделите и назовите классы]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(12005007008=\answerblank[68mm].\)
    \item Какая группа записывает класс миллионов? \answerblank[18mm].
    \item Сколько миллионов содержит число? \answerblank[18mm].
  \end{enumerate}
\end{practicebox}

\begin{practicebox}[2. Восстановите число]
  В числе \(4\) миллиарда, \(12\) миллионов, \(5\) тысяч и \(8\) единиц.
  \[
    4\;|\;\answerblank[16mm]\;|\;\answerblank[16mm]\;|\;
    \answerblank[16mm]=\answerblank[48mm].
  \]
\end{practicebox}

\begin{warningbox}[Проверьте группы]
  Группы справа от крайней левой должны содержать по три цифры. Поэтому числа
  классов \(12\), \(5\) и \(8\) записываются группами \(012\), \(005\) и
  \(008\).
\end{warningbox}

\section{Задачи на рассуждение}

\begin{practicebox}[3. Сравните по классам]
  Поставьте знак и назовите первый слева класс, в котором группы различаются.
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(5\,041\,008\,020\ \answerblank[8mm]\ 5\,041\,080\,002\).
    \item \(12\,007\,500\,090\ \answerblank[8mm]\ 12\,070\,005\,900\).
  \end{enumerate}
\end{practicebox}

\begin{practicebox}[4. Приведите два разных примера]
  Запишите два двенадцатизначных числа, у которых:
  \begin{itemize}
    \item в классе миллиардов записано \(305\);
    \item класс тысяч нулевой;
    \item в классе единиц записано \(042\);
    \item группы класса миллионов различаются.
  \end{itemize}
  Числа: \answerblank[48mm] и \answerblank[48mm].
\end{practicebox}

\begin{practicebox}[5. Обратная задача]
  Число записано четырьмя классами. В нём \(9\) миллиардов. Группа класса
  миллионов в \(3\) раза больше группы класса тысяч, а группа класса тысяч
  равна \(012\). Класс единиц нулевой.

  Найдите число и объясните построение.
  \writinglines{3}
\end{practicebox}

  \writinglines{2}
\end{reflectionbox}
