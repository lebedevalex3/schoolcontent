  {\modulemark{M03-R} Коррекционная карточка}

\studentnameblock

\begin{notebox}[Способ]
  Закройте число слева. Выделите последние три цифры, затем следующие три.
  Только крайняя левая группа может быть короче.
\end{notebox}

\section{Шаг 1. Действуем по частям}

Разделите число \(8305072\).
\begin{enumerate}[label=\textbf{\arabic*.}]
  \item Последние три цифры: \(\answerblank[20mm]\), класс единиц.
  \item Следующие три цифры: \(\answerblank[20mm]\), класс тысяч.
  \item Оставшаяся слева группа: \(\answerblank[16mm]\), класс миллионов.
\end{enumerate}
Итог: \(8305072=\answerblank[52mm].\)

\begin{practicebox}[Повторите способ]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(53008=\answerblank[42mm].\)
    \item \(70402009=\answerblank[50mm].\)
    \item \(302000070=\answerblank[54mm].\)
  \end{enumerate}
\end{practicebox}

\section{Шаг 3. Собираем число}

\begin{problembox}[Образец]
  \(8\) миллионов, \(3\) тысячи, \(9\) единиц:
  \[
    8\;|\;003\;|\;009=8\,003\,009.
  \]
\end{problembox}

\begin{practicebox}[Соберите самостоятельно]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(24\) миллиона, \(6\) тысяч, \(5\) единиц:
      \answerblank[55mm].
    \item \(7\) миллионов, \(0\) тысяч, \(80\) единиц:
      \answerblank[55mm].
  \end{enumerate}
\end{practicebox}

\begin{warningbox}[Исправьте две разные ошибки]
  \begin{enumerate}[label=\textbf{\arabic*.}]
    \item \(6150302=615\,|\,030\,|\,2\).
    \item \(42\) миллиона, \(7\) тысяч и \(9\) единиц записали как
      \(42\,7\,9\).
  \end{enumerate}
  Верные записи и причины ошибок:
  \writinglines{3}
\end{warningbox}
