\newcommand{\digitcard}[2][]{%
  \tikz[baseline=(n.base)]{%
    \node[
      draw=Primary,
      fill=PrimaryLight,
      rounded corners=1.2mm,
      minimum width=9mm,
      minimum height=10mm,
      inner sep=0pt,
      font=\bfseries\sffamily\large,
      #1
    ] (n) {#2};%
  }%
}
