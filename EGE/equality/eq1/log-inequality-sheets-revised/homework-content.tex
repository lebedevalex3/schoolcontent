\studentline

\begin{routebox}[Маршрут]
  Для полного решения придерживайтесь порядка:
  \begin{enumerate}
    \item найдите ОДЗ;
    \item преобразуйте чётные степени с помощью модуля;
    \item введите \(y=\log_b|f(x)|\), затем \(t=y^2\ge0\);
    \item решите квадратное неравенство;
    \item вернитесь от \(t\) к \(y\), затем к аргументу логарифма и
      исходной переменной.
  \end{enumerate}
\end{routebox}

\begin{homeworktask}[Различаем три похожие записи]{1}
  а) Расшифруйте:
  \[
    \lg^3 A=\answerblank[35mm],
    \qquad
    \lg(A^3)=\answerblank[35mm],
  \]
  \[
    \log_3 A=\answerblank[70mm].
  \]

  б) Вычислите оба значения:
  \[
    \lg^2 1000=\answerblank[25mm],
    \qquad
    \lg(1000^2)=\answerblank[25mm].
  \]

  в) Объясните, в чём разница между \(\lg^2 A\) и \(\lg(A^2)\).
  \answerspace{22mm}
\end{homeworktask}

\begin{homeworktask}[Исправляем формулу]{2}
  Ученик записал:
  \[
    \lg((x-1)^2)=2\lg(x-1).
  \]
  \begin{enumerate}
    \item На какой области определена левая часть?
    \item На какой области определена правая часть?
    \item Запишите равенство, верное для всех \(x\) из ОДЗ левой части.
  \end{enumerate}
  \answerspace{36mm}
\end{homeworktask}

\begin{homeworktask}[Решаем алгебраическое ядро]{3}
  Решите неравенство
  \[
    y^4-y^2-12\ge0.
  \]
  Введите \(t=y^2\). Какое ограничение на \(t\) следует из этой замены и
  почему его необходимо учесть?
  \[
    t^2-t-12=(t-\answerblank[12mm])(t+\answerblank[12mm]).
  \]
  \answerspace{35mm}
  \shortanswerline
\end{homeworktask}

\begin{homeworktask}[Решаем два неравенства с модулем]{4}
  Решите два неравенства. В первом случае сначала положите \(u=x-2\),
  во втором --- \(u=x^2-4\).
  \[
    \text{а) }0<|x-2|\le3,
    \qquad
    \text{б) }0<|x^2-4|\le1.
  \]
  Для каждого пункта изобразите множество решений на числовой прямой.
  \answerspace{62mm}
\end{homeworktask}

\begin{homeworktask}[Собираем решение с опорой]{5}
  Решите неравенство
  \[
    \log_2^4((x-1)^2)-\log_2^2((x-1)^4)\ge192.
  \]

  \structuredsolutionmap

  \begin{hintbox}[Контрольные ориентиры]
    Сначала найдите ОДЗ. Затем выберите выражение, которое удобно обозначить
    через \(y\), чтобы исходное неравенство превратилось в биквадратное.
    После преобразований должно получиться то же неравенство, что и в
    задании 3: используйте это как проверку.
  \end{hintbox}
\end{homeworktask}

\begin{homeworktask}[Выбираем подходящую замену]{6}
  Для выражения
  \[
    \lg^4((x^2-1)^2)-\lg^2((x^2-1)^4)
  \]
  предложены замены:
  \[
    \text{A. }t=\lg^2|x^2-1|,
    \qquad
    \text{Б. }t=\log_2|x^2-1|,
  \]
  \[
    \text{В. }t=\lg(|x^2-1|^2).
  \]
 Какая из предложенных замен сразу даёт двучлен второй степени
относительно \(t\)? Запишите её в виде \(t=(\ldots)^2\) ---
без верхнего индекса над \(\lg\).
  \answerspace{10mm}
\end{homeworktask}
